1049 explicit ParameterProperty_t(

1054
1060 std::string first() const
1061 {
1062 //Skip the size and return the string
1063 return std::string((char*)data + 4);
1064 }
1065
1071 std::string second() const
1072 {
1073 //Skip the first element
1074 uint32_t size1 = ParameterProperty_t::element_size(data);
1075
1076 //Skip the size of the second element and return the string
1077 return std::string((char*)data + size1 + 4);
1078 }
1079
1086 bool modify(
1087 const std::pair<std::string, std::string>& new_value)
1088 {
1089 uint32_t old_size = size();
1090
1091 uint32_t first_size = (uint32_t)new_value.first.size() + 1;
1092 uint32_t first_alignment = ((first_size + 3u) & ~3u) - first_size;
1093 uint32_t second_size = (uint32_t)new_value.second.size() + 1;
1094 uint32_t second_alignment = ((second_size + 3u) & ~3u) - second_size;
1095 uint32_t new_size = first_size + first_alignment + second_size + second_alignment + 8;
1096
1097 if (old_size != new_size)
1098 {
1099 return false;
1100 }
1101
1102 fastdds::rtps::octet* current = data;
1103 memcpy(current, &first_size, 4);
1104 memcpy(current + 4, new_value.first.c_str(), first_size);
1105 memset(current + 4 + first_size, 0, first_alignment);
1106
1107 current = data + 4 + first_size + first_alignment;
1108 memcpy(current, &second_size, 4);
1109 memcpy(current + 4, new_value.second.c_str(), second_size);
1110 memset(current + 4 + second_size, 0, second_alignment);
1111
1112 return true;
1113 }
1114
1120 std::pair<const std::string, const std::string> pair() const
1121 {
1122 return std::make_pair(std::string(first()), std::string(second()));
1123 }
1124

1131 {
1132 //Size of the first element (with alignment)
1133 uint32_t size1 = ParameterProperty_t::element_size(data);
1134
1135 //Size of the second element (with alignment)
1136 uint32_t size2 = ParameterProperty_t::element_size(data + size1);
1137 return size1 + size2;
1138 }

1145 }

1151 }
